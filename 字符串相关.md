#### 题目
##### 剑指 Offer 05. 替换空格
#### 地址
##### https://leetcode-cn.com/problems/ti-huan-kong-ge-lcof/
#### 方法一：string Builder
##### 复杂度分析
###### 时间复杂度：O(n)。
###### 空间复杂度：O(n)。
##### Golang实现
    func replaceSpace(s string) string {
    	var ss strings.Builder
    
    	for _, v := range s {
    		if v == ' ' {
    			ss.WriteString("%20")
    		} else {
    			ss.WriteRune(v)
    		}
    	}
    
    	return ss.String()
    }

#### 题目
##### 412. Fizz Buzz
#### 地址
##### https://leetcode-cn.com/problems/fizz-buzz/
#### 方法一：取余
##### 复杂度分析
###### 时间复杂度：O(n) 这里 n 是输入字符串的长度。
###### 空间复杂度：O(n)。
##### Golang实现
    func fizzBuzz(n int) []string {
        ret := make([]string, n)
    
        for i := 1; i <= n; i++ {
    		switch {
    		case i%15 == 0:
    			ret[i-1] = "FizzBuzz"
    		case i%3 == 0:
    			ret[i-1] = "Fizz"
    		case i%5 == 0:
    			ret[i-1] = "Buzz"
    		default:
    			ret[i-1] = strconv.Itoa(i)
    		}
    	}
    
    	return ret
    }
