#### 题目
##### 剑指 Offer 49. 丑数
#### 地址
##### https://leetcode-cn.com/problems/chou-shu-lcof/
#### 方法一：动态规划
##### 复杂度分析
###### 时间复杂度：O(N) 其中 N=n ，动态规划需遍历计算 dp 列表。
###### 空间复杂度：O(N) 长度为 N 的 dp 列表使用 O(N) 的额外空间。
##### Golang实现
    func nthUglyNumber(n int) int {
        ret := make([]int, n)
        ret[0] = 1
        i2, i3, i5 := 0, 0, 0
    
        for i := 1; i < n; i++ {
            ret[i] = min(ret[i2]*2, min(ret[i3]*3, ret[i5]*5))
    
            if ret[i] == ret[i2]*2 {
                i2++
            }
            if ret[i] == ret[i3]*3 {
                i3++
            }
            if ret[i] == ret[i5]*5 {
                i5++
            }
        }
    
        return ret[n-1]
    }
    
    func min(a, b int) int {
        if a > b {
            return b
        }
    
        return a
    }
