#### 题目
##### 283. 移动零
#### 地址
##### https://leetcode-cn.com/problems/move-zeroes/
#### 方法一：双指针
##### 复杂度分析
###### 时间复杂度：O(n)
###### 空间复杂度：O(1)
##### Golang实现
    func moveZeroes(nums []int) {
    	for i, j := 0, 0; i < len(nums); i++ {
    		if nums[i] != 0 {
    			nums[j], nums[i] = nums[i], nums[j]
    			j++
    		}
    	}
    }
##### Java实现
    class Solution {
        public void moveZeroes(int[] nums) {
            for(int i = 0, j = 0; i < nums.length; i++) {
                if(nums[i] != 0) {
                    int temp = nums[j];
                    nums[j] = nums[i];
                    nums[i] = temp;
                    j++;
                }
            }
        }
    }
